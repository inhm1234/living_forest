// V1.10.31 날짜별 집계 구조를 그대로 유지합니다.
// 목적: 관리자 대시보드에서 오늘 / 어제 / 최근 7일 / 전체 기준으로 안정적인 숫자를 보기 위한 전체 코드
// 참고: V1.10.31 이후 Apps Script가 정상 작동 중이라면 이번 V1.11에서 새로 배포하지 않아도 됩니다.
// 문제가 있을 때만 이 코드 전체를 붙여넣고 새 버전으로 웹 앱을 다시 배포하세요.

const SPREADSHEET_ID = "1EQlAMI_pK1DrUL7MC-efcneyLYKEHj0ipWJu077TDFI";
const PROJECT_KEY = "living_forest_v1";
const EVENT_SHEET_NAME = "events";
const SERVICE_TIME_ZONE = "Asia/Seoul";

const SUMMARY_RANGES = {
  today: "오늘",
  yesterday: "어제",
  last7: "최근 7일",
  all: "전체",
};

const TRACKED_EVENTS = [
  "app_opened",
  "first_visit_living_forest",
  "screen_view_world",
  "screen_view_garden",
  "go_garden_click",
  "tree_name_saved",
  "mood_recorded",
  "return_world_click",
  "share_click",
  "return_visit",
  "growth_milestone_reached",
  "growth_day_7_reached",
  "focus_my_tree_click",
];

function doGet(e) {
  const params = e && e.parameter ? e.parameter : {};
  const action = params.action || "";

  if (params.key !== PROJECT_KEY) {
    return outputJson_({ ok: false, error: "invalid_key" }, params.callback);
  }

  ensureSheet_();

  if (action === "track") {
    return trackEvent_(params);
  }

  if (action === "summary") {
    return outputJson_(buildSummary_(params), params.callback);
  }

  return outputJson_({ ok: false, error: "unknown_action" }, params.callback);
}

function trackEvent_(params) {
  const sheet = ensureSheet_();
  const now = new Date();
  const dateKey = Utilities.formatDate(now, SERVICE_TIME_ZONE, "yyyy-MM-dd");

  sheet.appendRow([
    now,
    dateKey,
    params.event_name || "",
    params.anon_id || "",
    params.app_version || "",
    params.schema || "",
    params.is_test_mode || "",
    params.page_path || "",
    params.growth_days || "",
    params.tree_stage || "",
    params.p_source || "",
    params.p_mood_type || "",
    params.p_milestone_day || "",
  ]);

  return outputJson_({ ok: true });
}

function buildSummary_(params) {
  const sheet = ensureSheet_();
  const values = sheet.getDataRange().getValues();
  const rangeInfo = getRangeInfo_(params.range || params.period || "today", params.date);
  const summary = createEmptySummary_();
  const visitorSet = {};
  let matchedRows = 0;

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowDateKey = normalizeRowDateKey_(row);

    if (!isDateInRange_(rowDateKey, rangeInfo)) {
      continue;
    }

    matchedRows += 1;

    const eventName = String(row[2] || "").trim();
    const anonId = String(row[3] || "").trim();

    if (anonId) {
      visitorSet[anonId] = true;
    } else if (isVisitEvent_(eventName)) {
      visitorSet[`row_${i}`] = true;
    }

    if (Object.prototype.hasOwnProperty.call(summary, eventName)) {
      summary[eventName] += 1;
    }
  }

  summary.visitors = Object.keys(visitorSet).length || summary.app_opened;

  return {
    ok: true,
    range: rangeInfo.range,
    rangeLabel: rangeInfo.label,
    date: rangeInfo.endDate || "전체",
    startDate: rangeInfo.startDate || "",
    endDate: rangeInfo.endDate || "",
    updatedAt: Utilities.formatDate(new Date(), SERVICE_TIME_ZONE, "yyyy-MM-dd HH:mm:ss"),
    summary,
    debug: {
      totalRows: Math.max(values.length - 1, 0),
      matchedRows,
    },
  };
}

function createEmptySummary_() {
  const summary = { visitors: 0 };

  TRACKED_EVENTS.forEach((eventName) => {
    summary[eventName] = 0;
  });

  return summary;
}

function getRangeInfo_(range, dateText) {
  const safeRange = Object.prototype.hasOwnProperty.call(SUMMARY_RANGES, range) ? range : "today";
  const today = Utilities.formatDate(new Date(), SERVICE_TIME_ZONE, "yyyy-MM-dd");
  const targetDate = normalizeDateKey_(dateText) || today;

  if (safeRange === "all") {
    return {
      range: "all",
      label: SUMMARY_RANGES.all,
      startDate: "",
      endDate: "",
    };
  }

  if (safeRange === "yesterday") {
    const yesterday = addDaysToDateKey_(targetDate, -1);
    return {
      range: "yesterday",
      label: SUMMARY_RANGES.yesterday,
      startDate: yesterday,
      endDate: yesterday,
    };
  }

  if (safeRange === "last7") {
    return {
      range: "last7",
      label: SUMMARY_RANGES.last7,
      startDate: addDaysToDateKey_(targetDate, -6),
      endDate: targetDate,
    };
  }

  return {
    range: "today",
    label: SUMMARY_RANGES.today,
    startDate: targetDate,
    endDate: targetDate,
  };
}

function isDateInRange_(dateKey, rangeInfo) {
  if (rangeInfo.range === "all") {
    return true;
  }

  if (!dateKey) {
    return false;
  }

  return dateKey >= rangeInfo.startDate && dateKey <= rangeInfo.endDate;
}

function isVisitEvent_(eventName) {
  return eventName === "app_opened" || eventName === "first_visit_living_forest" || eventName === "screen_view_world" || eventName === "screen_view_garden";
}

function normalizeRowDateKey_(row) {
  return normalizeDateKey_(row[1]) || normalizeDateKey_(row[0]);
}

function normalizeDateKey_(value) {
  if (value === null || value === undefined || value === "") {
    return "";
  }

  if (Object.prototype.toString.call(value) === "[object Date]" && !Number.isNaN(value.getTime())) {
    return Utilities.formatDate(value, SERVICE_TIME_ZONE, "yyyy-MM-dd");
  }

  if (typeof value === "number" && Number.isFinite(value)) {
    const excelEpoch = new Date(Date.UTC(1899, 11, 30));
    const dateValue = new Date(excelEpoch.getTime() + value * 24 * 60 * 60 * 1000);
    return Utilities.formatDate(dateValue, SERVICE_TIME_ZONE, "yyyy-MM-dd");
  }

  const text = String(value).trim();

  if (!text) {
    return "";
  }

  const directMatch = text.match(/^(\d{4})[-/.](\d{1,2})[-/.](\d{1,2})/);
  if (directMatch) {
    const year = directMatch[1];
    const month = String(Number(directMatch[2])).padStart(2, "0");
    const day = String(Number(directMatch[3])).padStart(2, "0");
    return `${year}-${month}-${day}`;
  }

  const compactMatch = text.match(/^(\d{4})(\d{2})(\d{2})$/);
  if (compactMatch) {
    return `${compactMatch[1]}-${compactMatch[2]}-${compactMatch[3]}`;
  }

  const parsed = new Date(text);
  if (!Number.isNaN(parsed.getTime())) {
    return Utilities.formatDate(parsed, SERVICE_TIME_ZONE, "yyyy-MM-dd");
  }

  return "";
}

function addDaysToDateKey_(dateKey, days) {
  const base = normalizeDateKey_(dateKey);
  if (!base) {
    return "";
  }

  const dateValue = new Date(`${base}T12:00:00+09:00`);
  const nextDateValue = new Date(dateValue.getTime() + days * 24 * 60 * 60 * 1000);
  return Utilities.formatDate(nextDateValue, SERVICE_TIME_ZONE, "yyyy-MM-dd");
}

function ensureSheet_() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(EVENT_SHEET_NAME);

  if (!sheet) {
    sheet = ss.insertSheet(EVENT_SHEET_NAME);
  }

  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "timestamp",
      "date",
      "event_name",
      "anon_id",
      "app_version",
      "schema",
      "is_test_mode",
      "page_path",
      "growth_days",
      "tree_stage",
      "source",
      "mood_type",
      "milestone_day",
    ]);
  }

  return sheet;
}

function outputJson_(data, callback) {
  const json = JSON.stringify(data);

  if (callback && /^[a-zA-Z_$][0-9a-zA-Z_$.]*$/.test(callback)) {
    return ContentService
      .createTextOutput(`${callback}(${json});`)
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
