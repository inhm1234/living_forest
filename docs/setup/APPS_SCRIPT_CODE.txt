const SPREADSHEET_ID = "1EQlAMI_pK1DrUL7MC-efcneyLYKEHj0ipWJu077TDFI";
const PROJECT_KEY = "living_forest_v1";

const EVENT_SHEET_NAME = "events";
const FRIEND_SEATS_SHEET_NAME = "friend_seats";
const FRIEND_LINKS_SHEET_NAME = "friend_links";
const SERVICE_TIME_ZONE = "Asia/Seoul";

function doGet(e) {
  const params = e && e.parameter ? e.parameter : {};
  const action = params.action || "";

  if (params.key !== PROJECT_KEY) {
    return outputJson_({ ok: false, error: "invalid_key" }, params.callback);
  }

  if (action === "track") {
    return trackEvent_(params);
  }

  if (action === "summary") {
    return outputJson_(buildSummary_(params), params.callback);
  }

  if (action === "join_friend_seat") {
    return outputJson_(joinFriendSeat_(params), params.callback);
  }

  if (action === "list_friend_seats") {
    return outputJson_(listFriendSeats_(params), params.callback);
  }

  if (action === "delete_friend_seat") {
    return outputJson_(deleteFriendSeat_(params), params.callback);
  }

  if (action === "list_friend_links") {
    return outputJson_(listFriendLinks_(params), params.callback);
  }

  if (action === "assign_friend_seat") {
    return outputJson_(assignFriendSeat_(params), params.callback);
  }

  if (action === "ping") {
    return outputJson_({ ok: true, service: "living_forest", version: "V1.55", timestamp: new Date().toISOString() }, params.callback);
  }

  return outputJson_({ ok: false, error: "unknown_action" }, params.callback);
}

/**
 * 기존 방문/클릭 기록 저장
 */
function trackEvent_(params) {
  const sheet = ensureEventSheet_();
  const now = new Date();
  const dateKey = Utilities.formatDate(now, SERVICE_TIME_ZONE, "yyyy-MM-dd");

  sheet.appendRow([
    now,
    dateKey,
    params.event_name || "",
    params.anon_id || "",
    params.app_version || "",
    params.schema || "",
    params.is_test_mode || "",
    params.page_path || "",
    params.growth_days || "",
    params.tree_stage || "",
    params.p_source || "",
    params.p_mood_type || "",
    params.p_milestone_day || "",
  ]);

  return outputJson_({ ok: true }, params.callback);
}

/**
 * 친구가 초대 링크로 들어와서 특정 숲의 특정 자리에 자기 나무를 저장하는 기능
 * V1.55부터 friend_seats와 friend_links를 함께 갱신합니다.
 */
function joinFriendSeat_(params) {
  const forestId = cleanText_(params.forest_id, 80);
  const seatId = cleanText_(params.seat_id, 40);
  const seatLabel = cleanText_(params.seat_label, 40);
  const friendId = cleanText_(params.friend_id, 80) || buildFallbackFriendId_();

  const treeName = cleanText_(params.tree_name, 40);
  const growthDays = cleanText_(params.growth_days, 10);
  const treeStage = cleanText_(params.tree_stage, 30);
  const moodLabel = cleanText_(params.mood_label, 30);
  const source = cleanText_(params.source, 40) || "friend_invite";

  if (!forestId) {
    return { ok: false, error: "missing_forest_id" };
  }

  if (!seatId) {
    return { ok: false, error: "missing_seat_id" };
  }

  if (!friendName) {
    return { ok: false, error: "missing_friend_name" };
  }

  const placeResult = placeFriendIntoSeat_({
    forestId,
    seatId,
    seatLabel,
    friendId,
    friendName,
    treeName,
    growthDays,
    treeStage,
    moodLabel,
    source,
    appVersion: params.app_version || "",
    isTestMode: params.is_test_mode || "",
  });

  return {
    ok: true,
    saved: true,
    forestId,
    forest_id: forestId,
    seatId,
    seat_id: seatId,
    seatLabel,
    seat_label: seatLabel,
    friendId,
    friend_id: friendId,
    friendName,
    friend_name: friendName,
    treeName,
    tree_name: treeName,
    growthDays,
    growth_days: growthDays,
    treeStage,
    tree_stage: treeStage,
    moodLabel,
    mood_label: moodLabel,
    status: "active",
    replaced: placeResult.replaced,
    displacedFriendId: placeResult.displacedFriendId,
    displaced_friend_id: placeResult.displacedFriendId,
    displacedFriendName: placeResult.displacedFriendName,
    displaced_friend_name: placeResult.displacedFriendName,
  };
}

/**
 * 내 숲 화면에서 해당 숲의 친구 자리 목록을 불러오는 기능
 */
function listFriendSeats_(params) {
  const sheet = ensureFriendSeatsSheet_();
  const forestId = cleanText_(params.forest_id, 80);

  if (!forestId) {
    return { ok: false, error: "missing_forest_id" };
  }

  const values = sheet.getDataRange().getValues();
  const seats = [];

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const status = String(row[14] || "active").trim();

    if (rowForestId !== forestId || status === "deleted") {
      continue;
    }

    seats.push(buildFriendSeatObject_(row));
  }

  return {
    ok: true,
    forestId,
    forest_id: forestId,
    count: seats.length,
    seats,
    updatedAt: Utilities.formatDate(new Date(), SERVICE_TIME_ZONE, "yyyy-MM-dd HH:mm:ss"),
  };
}

/**
 * 내 숲에서 특정 친구 자리를 비우는 기능
 * 데이터는 완전 삭제하지 않고 friend_seats의 status만 deleted로 바꿔 숨깁니다.
 * friend_links의 친구 관계는 남기고, placement_status만 unplaced로 바꿉니다.
 */
function deleteFriendSeat_(params) {
  const sheet = ensureFriendSeatsSheet_();
  const forestId = cleanText_(params.forest_id, 80);
  const seatId = cleanText_(params.seat_id, 40);

  if (!forestId) {
    return { ok: false, error: "missing_forest_id" };
  }

  if (!seatId) {
    return { ok: false, error: "missing_seat_id" };
  }

  const values = sheet.getDataRange().getValues();
  let targetRow = -1;
  let friendId = "";
  let friendName = "";

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const rowSeatId = String(row[3] || "").trim();
    const status = String(row[14] || "active").trim();

    if (rowForestId === forestId && rowSeatId === seatId && status !== "deleted") {
      targetRow = i + 1;
      friendId = String(row[5] || "");
      friendName = String(row[6] || "");
      break;
    }
  }

  if (targetRow < 0) {
    return { ok: false, error: "seat_not_found" };
  }

  const now = new Date();
  sheet.getRange(targetRow, 2).setValue(now);                 // updated_at
  sheet.getRange(targetRow, 12).setValue("owner_clear_seat"); // source
  sheet.getRange(targetRow, 15).setValue("deleted");          // status

  if (friendId) {
    setFriendLinkPlacement_(forestId, friendId, "", "", "unplaced", "owner_clear_seat");
  }

  return {
    ok: true,
    deleted: true,
    forestId,
    forest_id: forestId,
    seatId,
    seat_id: seatId,
    friendId,
    friend_id: friendId,
    friendName,
    friend_name: friendName,
    status: "deleted",
  };
}

/**
 * V1.55: 친구 관계 목록 불러오기
 */
function listFriendLinks_(params) {
  const forestId = cleanText_(params.forest_id, 80);

  if (!forestId) {
    return { ok: false, error: "missing_forest_id" };
  }

  migrateFriendLinksFromActiveSeats_(forestId, params);

  const sheet = ensureFriendLinksSheet_();
  const values = sheet.getDataRange().getValues();
  const links = [];

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const status = String(row[13] || "active").trim();

    if (rowForestId !== forestId || status === "deleted") {
      continue;
    }

    links.push(buildFriendLinkObject_(row));
  }

  links.sort(function(a, b) {
    return String(b.updatedAt || b.updated_at || "").localeCompare(String(a.updatedAt || a.updated_at || ""));
  });

  return {
    ok: true,
    forestId,
    forest_id: forestId,
    count: links.length,
    links,
    updatedAt: Utilities.formatDate(new Date(), SERVICE_TIME_ZONE, "yyyy-MM-dd HH:mm:ss"),
  };
}

/**
 * V1.55: 친구 목록에 있는 친구를 원하는 자리로 배치/이동/교체
 */
function assignFriendSeat_(params) {
  const forestId = cleanText_(params.forest_id, 80);
  const friendId = cleanText_(params.friend_id, 80);
  const seatId = cleanText_(params.seat_id, 40);
  const seatLabel = cleanText_(params.seat_label, 40);
  const source = cleanText_(params.source, 40) || "owner_assign_friend_link";

  if (!forestId) {
    return { ok: false, error: "missing_forest_id" };
  }

  if (!friendId) {
    return { ok: false, error: "missing_friend_id" };
  }

  if (!seatId) {
    return { ok: false, error: "missing_seat_id" };
  }

  migrateFriendLinksFromActiveSeats_(forestId, params);

  const linkRow = findFriendLinkRow_(forestId, friendId);
  if (!linkRow) {
    return { ok: false, error: "friend_link_not_found" };
  }

  const link = buildFriendLinkObject_(linkRow.row);
  const result = placeFriendIntoSeat_({
    forestId,
    seatId,
    seatLabel,
    friendId,
    friendName: link.friendName || link.friend_name || "친구",
    treeName: link.treeName || link.tree_name || "친구 나무",
    growthDays: link.growthDays || link.growth_days || "0",
    treeStage: link.treeStage || link.tree_stage || "germination",
    moodLabel: link.moodLabel || link.mood_label || "기록 전",
    source,
    appVersion: params.app_version || "",
    isTestMode: params.is_test_mode || "",
  });

  return {
    ok: true,
    assigned: true,
    forestId,
    forest_id: forestId,
    friendId,
    friend_id: friendId,
    seatId,
    seat_id: seatId,
    seatLabel,
    seat_label: seatLabel,
    replaced: result.replaced,
    displacedFriendId: result.displacedFriendId,
    displaced_friend_id: result.displacedFriendId,
    displacedFriendName: result.displacedFriendName,
    displaced_friend_name: result.displacedFriendName,
    link: result.link,
    seat: result.seat,
  };
}

function placeFriendIntoSeat_(data) {
  const sheet = ensureFriendSeatsSheet_();
  const now = new Date();
  const values = sheet.getDataRange().getValues();
  let targetRow = -1;
  let replaced = false;
  let displacedFriendId = "";
  let displacedFriendName = "";
  let displacedRow = null;
  const rowsToDelete = [];

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const rowSeatId = String(row[3] || "").trim();
    const rowFriendId = String(row[5] || "").trim();
    const status = String(row[14] || "active").trim();

    if (rowForestId !== data.forestId) {
      continue;
    }

    if (rowSeatId === data.seatId) {
      targetRow = i + 1;
      if (status !== "deleted" && rowFriendId && rowFriendId !== data.friendId) {
        replaced = true;
        displacedFriendId = rowFriendId;
        displacedFriendName = String(row[6] || "");
        displacedRow = row;
      }
    }

    if (status !== "deleted" && rowFriendId === data.friendId && rowSeatId !== data.seatId) {
      rowsToDelete.push(i + 1);
    }
  }

  rowsToDelete.forEach(function(rowNumber) {
    sheet.getRange(rowNumber, 2).setValue(now);
    sheet.getRange(rowNumber, 12).setValue("moved_to_other_seat");
    sheet.getRange(rowNumber, 15).setValue("deleted");
  });

  if (displacedFriendId && displacedFriendId !== data.friendId) {
    if (displacedRow) {
      upsertFriendLink_({
        forestId: data.forestId,
        friendId: displacedFriendId,
        friendName: displacedFriendName,
        treeName: String(displacedRow[7] || ""),
        growthDays: String(displacedRow[8] || ""),
        treeStage: String(displacedRow[9] || ""),
        moodLabel: String(displacedRow[10] || ""),
        source: "replaced_by_other_friend",
        appVersion: data.appVersion || String(displacedRow[12] || ""),
        isTestMode: data.isTestMode || String(displacedRow[13] || ""),
        currentSeatId: "",
        currentSeatLabel: "",
        placementStatus: "unplaced",
      });
    } else {
      setFriendLinkPlacement_(data.forestId, displacedFriendId, "", "", "unplaced", "replaced_by_other_friend");
    }
  }

  const rowData = [
    now,
    now,
    data.forestId,
    data.seatId,
    data.seatLabel,
    data.friendId,
    data.friendName,
    data.treeName,
    data.growthDays,
    data.treeStage,
    data.moodLabel,
    data.source,
    data.appVersion,
    data.isTestMode,
    "active",
  ];

  if (targetRow > 0) {
    sheet.getRange(targetRow, 1, 1, rowData.length).setValues([rowData]);
  } else {
    sheet.appendRow(rowData);
  }

  const link = upsertFriendLink_({
    forestId: data.forestId,
    friendId: data.friendId,
    friendName: data.friendName,
    treeName: data.treeName,
    growthDays: data.growthDays,
    treeStage: data.treeStage,
    moodLabel: data.moodLabel,
    source: data.source,
    appVersion: data.appVersion,
    isTestMode: data.isTestMode,
    currentSeatId: data.seatId,
    currentSeatLabel: data.seatLabel,
    placementStatus: "placed",
  });

  return {
    replaced,
    displacedFriendId,
    displacedFriendName,
    link,
    seat: buildFriendSeatObject_(rowData),
  };
}

function upsertFriendLink_(data) {
  const sheet = ensureFriendLinksSheet_();
  const now = new Date();
  const values = sheet.getDataRange().getValues();
  let targetRow = -1;
  let originalTimestamp = now;
  let originalSource = data.source || "";

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const rowFriendId = String(row[3] || "").trim();

    if (rowForestId === data.forestId && rowFriendId === data.friendId) {
      targetRow = i + 1;
      originalTimestamp = row[0] || now;
      originalSource = row[9] || data.source || "";
      break;
    }
  }

  const rowData = [
    originalTimestamp,
    now,
    data.forestId,
    data.friendId,
    data.friendName,
    data.treeName,
    data.growthDays,
    data.treeStage,
    data.moodLabel,
    originalSource,
    data.source,
    data.appVersion,
    data.isTestMode,
    "active",
    data.currentSeatId || "",
    data.currentSeatLabel || "",
    data.placementStatus || "unplaced",
  ];

  if (targetRow > 0) {
    sheet.getRange(targetRow, 1, 1, rowData.length).setValues([rowData]);
  } else {
    sheet.appendRow(rowData);
  }

  return buildFriendLinkObject_(rowData);
}

function setFriendLinkPlacement_(forestId, friendId, seatId, seatLabel, placementStatus, source) {
  const sheet = ensureFriendLinksSheet_();
  const values = sheet.getDataRange().getValues();
  const now = new Date();

  for (let i = 1; i < values.length; i += 1) {
    const rowForestId = String(values[i][2] || "").trim();
    const rowFriendId = String(values[i][3] || "").trim();

    if (rowForestId === forestId && rowFriendId === friendId) {
      const rowNumber = i + 1;
      sheet.getRange(rowNumber, 2).setValue(now);                 // updated_at
      sheet.getRange(rowNumber, 11).setValue(source || "");       // latest_source
      sheet.getRange(rowNumber, 15).setValue(seatId || "");       // current_seat_id
      sheet.getRange(rowNumber, 16).setValue(seatLabel || "");    // current_seat_label
      sheet.getRange(rowNumber, 17).setValue(placementStatus || "unplaced");
      return true;
    }
  }

  return false;
}

function findFriendLinkRow_(forestId, friendId) {
  const sheet = ensureFriendLinksSheet_();
  const values = sheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const rowFriendId = String(row[3] || "").trim();
    const status = String(row[13] || "active").trim();

    if (rowForestId === forestId && rowFriendId === friendId && status !== "deleted") {
      return { rowNumber: i + 1, row };
    }
  }

  return null;
}

function migrateFriendLinksFromActiveSeats_(forestId, params) {
  const seatSheet = ensureFriendSeatsSheet_();
  ensureFriendLinksSheet_();

  const values = seatSheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    const rowForestId = String(row[2] || "").trim();
    const status = String(row[14] || "active").trim();

    if (rowForestId !== forestId || status === "deleted") {
      continue;
    }

    const friendId = String(row[5] || "").trim();
    if (!friendId) {
      continue;
    }

    upsertFriendLink_({
      forestId,
      friendId,
      friendName: String(row[6] || ""),
      treeName: String(row[7] || ""),
      growthDays: String(row[8] || ""),
      treeStage: String(row[9] || ""),
      moodLabel: String(row[10] || ""),
      source: "migrate_from_friend_seats",
      appVersion: params.app_version || String(row[12] || ""),
      isTestMode: params.is_test_mode || String(row[13] || ""),
      currentSeatId: String(row[3] || ""),
      currentSeatLabel: String(row[4] || ""),
      placementStatus: "placed",
    });
  }
}

/**
 * 기존 관리자 요약
 */
function buildSummary_(params) {
  const sheet = ensureEventSheet_();
  const values = sheet.getDataRange().getValues();
  const today = Utilities.formatDate(new Date(), SERVICE_TIME_ZONE, "yyyy-MM-dd");

  const summary = {
    visitors: 0,
    app_opened: 0,
    first_visit_living_forest: 0,
    screen_view_world: 0,
    screen_view_garden: 0,
    go_garden_click: 0,
    tree_name_saved: 0,
    mood_recorded: 0,
    return_world_click: 0,
    share_click: 0,
    return_visit: 0,
    growth_day_7_reached: 0,
    focus_my_tree_click: 0,
    friend_invite_seat_selected: 0,
    friend_invite_preview_opened: 0,
    friend_invite_link_copied: 0,
    friend_invite_kakao_shared: 0,
    online_friend_seat_joined: 0,
    online_friend_seat_deleted: 0,
    online_friend_link_assigned: 0,
  };

  const visitorSet = {};
  let totalRows = 0;
  let countedRows = 0;

  for (let i = 1; i < values.length; i += 1) {
    const row = values[i];
    totalRows += 1;

    const eventName = String(row[2] || "").trim();
    const anonId = String(row[3] || `row_${i}`);

    if (!eventName) {
      continue;
    }

    countedRows += 1;

    if (
      eventName === "app_opened" ||
      eventName === "first_visit_living_forest" ||
      eventName === "screen_view_world"
    ) {
      visitorSet[anonId] = true;
    }

    if (Object.prototype.hasOwnProperty.call(summary, eventName)) {
      summary[eventName] += 1;
    }
  }

  summary.visitors =
    Object.keys(visitorSet).length ||
    summary.app_opened ||
    summary.screen_view_world;

  return {
    ok: true,
    date: today,
    updatedAt: Utilities.formatDate(new Date(), SERVICE_TIME_ZONE, "yyyy-MM-dd HH:mm:ss"),
    debug: {
      totalRows,
      countedRows,
      note: "현재는 날짜 필터 없이 전체 events 행 기준으로 집계 중",
    },
    summary,
  };
}

/**
 * events 시트 준비
 */
function ensureEventSheet_() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(EVENT_SHEET_NAME);

function ensureHeaders_(sheet, headers) {
  const width = headers.length;
  const current = sheet.getRange(1, 1, 1, width).getValues()[0];
  let needsUpdate = false;

  for (let i = 0; i < width; i += 1) {
    if (String(current[i] || "").trim() !== headers[i]) {
      needsUpdate = true;
      break;
    }
  }

  if (needsUpdate) {
    sheet.getRange(1, 1, 1, width).setValues([headers]);
  }
}

/**
 * 텍스트 안전 정리
 */
function cleanText_(value, maxLength) {
  return String(value || "")
    .replace(/[<>]/g, "")
    .trim()
    .slice(0, maxLength || 100);
}

function buildFallbackFriendId_() {
  return "friend_" + Utilities.getUuid().replace(/-/g, "").slice(0, 16);
}

/**
 * 날짜를 문자열로 변환
 */
function valueToString_(value) {
  if (value instanceof Date) {
    return Utilities.formatDate(value, SERVICE_TIME_ZONE, "yyyy-MM-dd HH:mm:ss");
  }

  return String(value || "");
}

/**
 * JSON / JSONP 응답
 */
function outputJson_(data, callback) {
  const json = JSON.stringify(data);

  if (callback && /^[a-zA-Z_$][0-9a-zA-Z_$.]*$/.test(callback)) {
    return ContentService
      .createTextOutput(`${callback}(${json});`)
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
